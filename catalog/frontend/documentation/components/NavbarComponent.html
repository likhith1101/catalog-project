import { Component } from '@angular/core';
import { AuthService } from '../services/auth.service';
import { Router } from '@angular/router';

@Component({
  selector: 'app-navbar',
  templateUrl: './navbar.component.html',
  styleUrls: ['./navbar.component.css']
})
export class NavbarComponent {


  constructor(public authService: AuthService, private router: Router){}

  isMenuOpen: boolean = false;

  toggleMenu() {
    this.isMenuOpen = !this.isMenuOpen;
  }

  logout() {
    // Perform logout actions if necessary
    this.authService.logout();
    // Redirect to the login page
    this.router.navigate(['/login']);
  }
  
}

  <div class="navbar">

          <li>
            <a routerLink="/feature" class="nav-link" routerLinkActive="active">Feature</a>

          <li>
            <a routerLink="/parameter" class="nav-link" routerLinkActive="active">Parameter</a>
          </li>
          <li>
            <a [routerLink]="authService.isUser() ? '/summary' : '/login'" class="nav-link" routerLinkActive="active">Summary</a>
          </li>
        </ng-container>

        <!-- User Links -->
        <ng-container *ngIf="authService.isUser(); ">
          <li>
            <a routerLink="/summary" class="nav-link" routerLinkActive="active">Summary</a>
          </li>
        </ng-container>

        <!-- Default: Redirect to Login -->
        <li *ngIf="!authService.isLoggedIn">
          <a routerLink="/login" class="nav-link" routerLinkActive="active">Login</a>
        </li>

        <!-- Logout Link -->
        <li *ngIf="authService.isLoggedIn">
          <a (click)="logout()" class="nav-link">Logout</a>
        </li>
      </ul>
    </nav>
  </div>
</header>
